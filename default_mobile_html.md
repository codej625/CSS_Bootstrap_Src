# 기본적인 모바일 페이지 레이아웃을 만들어 보자!

<br />

```html
<!DOCTYPE html>
<html lang="ko">
  <head>
    <meta charset="utf-8"><!-- 인코딩 방식 설정 -->
    <meta name="Referrer" content="origin">
    <meta http-equiv="X-UA-Compatible" content="IE=edge"><!-- 최신 버전의 Internet Explorer를 사용하도록 브라우저를 강제하는 역할 -->
    <meta http-equiv="cache-control" content="no-cache"><!-- 브라우저가 페이지를 캐시하지 않도록 설정 -->
    <meta name="format-detection" content="telephone=no"><!-- 브라우저가 문서 내에서 전화번호를 자동으로 감지하지 않도록 설정 -->
    <meta name="viewport" content="width=device-width, minimum-scale=1, initial-scale=1, maximum-scale=1, user-scalable=no"><!-- 뷰포트 설정 -->
    <meta name="description" content=""><!-- 웹 페이지의 요약 정보 -->
    <meta name="keywords" content=""><!-- 페이지를 설명하는 키워드 -->
    <meta name="author" content=""><!-- 웹 페이지의 저자 정보 -->
    <!-- OG -->
    <meta property="og:type" content="">
    <meta property="og:title" content="">
    <meta property="og:description" content="">
    <meta property="og:image" content="">
    <meta property="og:url" content="">
    <!-- Favicon -->
    <link rel="icon" type="image/png" sizes="32x32" href="/img/common/favicon/favicon-32x32.png">
    <link rel="icon" type="image/png" sizes="16x16" href="/img/common/favicon/favicon-16x16.png">
    <link rel="manifest" href="/img/common/favicon/manifest.json">
    <!-- CSS -->
    <title></title>
  </head>
  
  <body>
    <div id="wrap">
      <header>
        <nav></nav>
      </header>
  
      <main>
        <section></section>
      </main>
  
      <footer></footer>
    <div>
  
    <!-- SCRIPT -->
  </body>
</html>
